Example

Create a HTTPS server on port 443 

https("./routes.test.js", 443, {
 privateKey: "./path/to/privateKey",
 certificate: "./path/to/certificate"
}, {
  views: "views",
  viewEngine: "ejs",
  static: "static",
  session: {
    use: true,
    name: "sessionId",
    secret: "unknown string",
    resave: false,
    saveUninitialized: true
  },
  authFunction: function(req, res, next) {
    next();
  }
});
